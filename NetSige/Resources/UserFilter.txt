SELECT
    USER.ID,
    CASE USER.STATUSID
        WHEN 0 THEN 'Ativo'
        WHEN 1 THEN 'Inativo'
    END AS 'Status',
    STRFTIME('%d/%m/%Y',USER.CREATIONDATE) As 'Data',
    USER.NAME AS 'Nome',
    USER.USERNAME AS 'Usuário',
    USER.PHONE AS 'Telefone',
    USER.EMAIL AS 'E-Mail'
FROM USER
WHERE
    USER.ID <> 1 AND
    IFNULL(USER.ID, '') LIKE @ID AND
    IFNULL(USER.STATUSID, '') LIKE @STATUSID AND
    IFNULL(USER.NAME, '') LIKE @NAME AND
    IFNULL(USER.USERNAME, '') LIKE @USERNAME AND
    IFNULL(USER.PHONE, '') LIKE @PHONE AND
    IFNULL(USER.EMAIL, '') LIKE @EMAIL AND
    USER.CREATIONDATE BETWEEN @CREATIONDATEI AND @CREATIONDATEF;

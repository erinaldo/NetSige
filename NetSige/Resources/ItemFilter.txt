SELECT
    ITEM.ID,
    CASE ITEM.STATUSID
        WHEN 0 THEN 'Ativo'
        WHEN 1 THEN 'Inativo'
    END AS 'Status',
    STRFTIME('%d/%m/%Y', ITEM.CREATIONDATE) AS 'Data',
    ITEM.NAME AS 'Nome',
    ITEMUNIT.INITIALS AS 'Unidade',
    ITEMGROUP.NAME AS 'Grupo',
    ITEM.INTERNALCODE AS 'Cód. Interno',
    ITEM.MANUFACTURER AS 'Fabricante',
    ITEM.MANUFACTURERCODE AS 'Cód. Fabricante',
    ITEM.NOTE AS 'Observação'
FROM ITEM
LEFT JOIN ITEMUNIT ON ITEMUNIT.ID = ITEM.ITEMUNITID
LEFT JOIN ITEMGROUP ON ITEMGROUP.ID = ITEM.ITEMGROUPID
WHERE
    IFNULL(ITEM.ID, '') LIKE @ID AND
    IFNULL(ITEM.STATUSID, '') LIKE @STATUSID AND
    IFNULL(ITEM.NAME, '') LIKE @NAME AND
    IFNULL(ITEMUNIT.INITIALS, '') LIKE @ITEMUNIT AND
    IFNULL(ITEMGROUP.NAME, '') LIKE @ITEMGROUP AND
    IFNULL(ITEM.INTERNALCODE, '') LIKE @INTERNALCODE AND
    IFNULL(ITEM.MANUFACTURER, '') LIKE @MANUFACTURER AND
    IFNULL(ITEM.MANUFACTURERCODE, '') LIKE @MANUFACTURERCODE AND
    IFNULL(ITEM.NOTE, '') LIKE @NOTE AND
    ITEM.CREATIONDATE BETWEEN @CREATIONDATEI AND @CREATIONDATEF;

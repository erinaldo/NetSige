SELECT
    PAYMENTTERM.ID,
    CASE PAYMENTTERM.STATUSID
        WHEN 0 THEN 'Ativo'
        WHEN 1 THEN 'Inativo'
    END AS 'Status',
    STRFTIME('%d/%m/%Y', PAYMENTTERM.CREATIONDATE) As 'Data',
    PAYMENTTERM.NAME AS 'Nome',
    CASE PAYMENTTERM.TYPEOFDAYSID
        WHEN 0 THEN 'Dias Úteis'
        WHEN 1 THEN 'Dias Corridos'
    END AS 'Tipo de Prazo'
FROM PAYMENTTERM
WHERE
    IFNULL(PAYMENTTERM.ID, '') LIKE @ID AND
    IFNULL(PAYMENTTERM.STATUSID, '') LIKE @STATUSID AND
    IFNULL(PAYMENTTERM.NAME, '') LIKE @NAME AND
    IFNULL(PAYMENTTERM.TYPEOFDAYSID, '') LIKE @TYPEOFDAYSID AND
    PAYMENTTERM.CREATIONDATE BETWEEN @CREATIONDATEI AND @CREATIONDATEF;

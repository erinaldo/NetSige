SELECT
    PURCHASEORDERITEM.ID,
    PURCHASEORDERITEM.SEQ,
    PURCHASEORDERITEM.DELIVERYDATE,
    PURCHASEORDERITEM.PURCHASEORDERID,
    PURCHASEORDERITEM.ITEMID,
    PURCHASEORDERITEM.ITEMUNITID,
    PURCHASEORDERITEM.QUANTITY,
    ITEMUNIT.INITIALS AS 'UNITNAME',
    ITEM.NAME AS 'ITEMNAME',
    PURCHASEORDERITEM.PRICE,
    PURCHASEORDERITEM.DISCOUNT,
    PURCHASEORDERITEM.COMPLEMENT,
    PURCHASEORDERITEM.IPI,
    PURCHASEORDERITEM.ICMS
FROM PURCHASEORDERITEM
LEFT JOIN ITEMUNIT ON ITEMUNIT.ID = PURCHASEORDERITEM.ITEMUNITID
LEFT JOIN ITEM ON ITEM.ID = PURCHASEORDERITEM.ITEMID
WHERE PURCHASEORDERITEM.PURCHASEORDERID = @ID;

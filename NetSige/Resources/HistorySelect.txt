SELECT
    USER.USERNAME AS 'Usuário',
    HISTORY.FIELD AS 'Campo',
    HISTORY.OLDVALUE AS 'Valor Antigo',
    HISTORY.NEWVALUE AS 'Valor Novo',
    STRFTIME('%d/%m/%Y às %H:%i:%s', HISTORY.CHANGEDATE) AS 'Data'
FROM HISTORY
LEFT JOIN USER ON USER.ID = HISTORY.USERID
WHERE HISTORY.ROUTINEID = @ROUTINEID
    AND HISTORY.REGISTRY = @REGISTRY;

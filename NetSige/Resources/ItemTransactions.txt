SELECT
    PURCHASEORDER.ID PURCHASEORDER,
    STRFTIME('%d/%m/%Y', PURCHASEORDER.CREATIONDATE) As 'CREATIONDATE',
    PERSON.NAME PERSON,
    ITEMUNIT.INITIALS UNIT,
    PURCHASEORDERITEM.QUANTITY/100 AS QUANTITY,
    PURCHASEORDERITEM.PRICE/100 AS PRICE,
    CAST(PURCHASEORDERITEM.IPI AS REAL)/100 AS IPI
FROM PURCHASEORDERITEM
LEFT JOIN PURCHASEORDER ON PURCHASEORDER.ID = PURCHASEORDERITEM.PURCHASEORDERID
LEFT JOIN PERSON ON PERSON.ID = PURCHASEORDER.PERSONID
LEFT JOIN ITEM ON ITEM.ID = PURCHASEORDERITEM.ITEMID
LEFT JOIN ITEMUNIT ON ITEMUNIT.ID = PURCHASEORDERITEM.ITEMUNITID
WHERE ITEM.ID = @ID
ORDER BY PURCHASEORDER.CREATIONDATE DESC;

UPDATE COSTSHARINGLOCAL SET
    NAME = @NAME,
    ZIPCODE = @ZIPCODE,
    ADDRESS = @ADDRESS,
    NUMBER = @NUMBER,
    COMPLEMENT = @COMPLEMENT,
    DISTRICT = @DISTRICT,
    CITY = @CITY,
    STATE = @STATE,
    PHONE = @PHONE,
    EMAIL = @EMAIL
WHERE COSTSHARINGLOCAL.COSTSHARINGID = @COSTSHARINGID
     AND COSTSHARINGLOCAL.ID = @ID;

UPDATE PURCHASEORDERITEM SET
    ITEMID = @ITEMID,
    QUANTITY = @QUANTITY,
    ITEMUNITID = @ITEMUNITID,
    PRICE = @PRICE,
    DISCOUNT = @DISCOUNT,
    COMPLEMENT = @COMPLEMENT,
    IPI = @IPI,
    ICMS = @ICMS
WHERE PURCHASEORDERITEM.PURCHASEORDERID = @PURCHASEORDERID
    AND PURCHASEORDERITEM.ID = @ID;

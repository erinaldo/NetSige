SELECT
    ITEMGROUP.ID,
    CASE ITEMGROUP.STATUSID
        WHEN 0 THEN 'Ativo'
        WHEN 1 THEN 'Inativo'
    END AS 'Status',
    STRFTIME('%d/%m/%Y', ITEMGROUP.CREATIONDATE) As 'Data',
    ITEMGROUP.NAME AS 'Nome'
FROM ITEMGROUP
WHERE
    IFNULL(ITEMGROUP.ID, '') LIKE @ID AND
    IFNULL(ITEMGROUP.STATUSID, '') LIKE @STATUSID AND
    IFNULL(ITEMGROUP.NAME, '') LIKE @NAME AND
    ITEMGROUP.CREATIONDATE BETWEEN @CREATIONDATEI AND @CREATIONDATEF;

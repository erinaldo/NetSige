SELECT
    ITEMUNIT.ID,
    CASE ITEMUNIT.STATUSID
        WHEN 0 THEN 'Ativo'
        WHEN 1 THEN 'Inativo'
    END AS 'Status',
    STRFTIME('%d/%m/%Y', ITEMUNIT.CREATIONDATE) As 'Data',
    ITEMUNIT.NAME AS 'Nome',
    ITEMUNIT.INITIALS AS 'Sigla'
FROM ITEMUNIT
WHERE
    IFNULL(ITEMUNIT.ID, '') LIKE @ID AND
    IFNULL(ITEMUNIT.STATUSID, '') LIKE @STATUSID AND
    IFNULL(ITEMUNIT.NAME, '') LIKE @NAME AND
    IFNULL(ITEMUNIT.INITIALS, '') LIKE @INITIALS AND
    ITEMUNIT.CREATIONDATE BETWEEN @CREATIONDATEI AND @CREATIONDATEF;
